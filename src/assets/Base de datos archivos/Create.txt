CREATE TABLE EMPRESA(
 ID_empresa VARCHAR2(7),
 Nombre VARCHAR2(40),
 Correo VARCHAR2(40),
 Contra VARCHAR2(30),
 Telefono VARCHAR2(11),
 Pais VARCHAR2(15),
 Fecha_alta DATE,
 CONSTRAINT PK_EMP PRIMARY KEY (ID_EMPRESA)
);

create table Restaurante(
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Nombre VARCHAR2(20),
    Descripcion VARCHAR2(90),
    Tipo VARCHAR2(25),
    Logo VARCHAR2(50),
    Ciudad VARCHAR2(20),
    Calle VARCHAR2(40),
    Fraccionamiento VARCHAR2(40),
    NumExt NUMBER(4),
    CP NUMBER(7),
    EmpresaID VARCHAR2(7),
    Latitud VARCHAR2(25),
    Longitud VARCHAR2(25),
    CONSTRAINT PK_RES PRIMARY KEY(ID),
    CONSTRAINT FK_EmDir FOREIGN KEY (EmpresaID) REFERENCES Empresa(ID_empresa)
)
